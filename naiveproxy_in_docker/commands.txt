# create container
docker compose build --no-cache && docker compose up

# in container

# build naive
cd /naiveproxy/src; ./get-clang.sh; ccache -z && ./build.sh && ccache -s

# run naive with config.json
cd /naiveproxy; nohup ./src/out/Release/naive > naive.log 2>&1 &

# logs
tail -20 naive.log

# check is running
pgrep -l naive

# stop naive
pkill naive

# For android apk (needs ANDROID_HOME — already set in Dockerfile)
cd /naiveproxy/src; export EXTRA_FLAGS='target_cpu="arm64" target_os="android"'; ./get-clang.sh && ./build.sh; mkdir -p /naiveproxy/apk/app/libs/arm64-v8a; cp out/Release/android/naive /naiveproxy/apk/app/libs/arm64-v8a/libnaive.so; cd /naiveproxy/apk; export APK_ABI=arm64-v8a; export APK_VERSION_NAME=v1.0.0-1; ./gradlew :app:assembleDebug
